        // --- IMPLEMENTAZIONE FETCH CON RETRY ED ESPONENTIAL BACKOFF ---
        async function fetchWithBackoff(url, options = {}, retries = 5, delay = 1000) {
            try {
                const response = await fetch(url, options);
                if (!response.ok) {
                    throw new Error(`Errore HTTP! Stato: ${response.status}`);
                }
                return await response.json();
            } catch (error) {
                if (retries > 1) {
                    await new Promise(resolve => setTimeout(resolve, delay));
                    return fetchWithBackoff(url, options, retries - 1, delay * 2);
                } else {
                    throw error;
                }
            }
        }

        // --- CALCOLO BOUNDING BOX AEREO PER LE QUERY ---
        function getBoundingBox(lat, lng, radiusKm) {
            const latSpan = radiusKm / 111.32;
            const lngSpan = radiusKm / (111.32 * Math.cos(lat * Math.PI / 180));
            return {
                lamin: lat - latSpan,
                lamax: lat + latSpan,
                lomin: lng - lngSpan,
                lomax: lng + lngSpan
            };
        }

        // --- FETCH DATI RADAR (AirLabs + OpenSky) ---
        async function fetchRadarData() {
            const loader = document.getElementById('radar-loader');
            const iconReady = document.getElementById('radar-icon-ready');
            
            loader.classList.remove('hidden');
            iconReady.classList.add('hidden');

            const lat = parseFloat(document.getElementById('input-center-lat').value);
            const lng = parseFloat(document.getElementById('input-center-lng').value);
            const radiusKm = parseFloat(document.getElementById('slider-radius').value);

            if (isNaN(lat) || isNaN(lng)) {
                alertMessage("Inserisci coordinate di partenza valide.");
                loader.classList.add('hidden');
                iconReady.classList.remove('hidden');
                return;
            }

            // Ottieni i confini del bounding box in base al raggio impostato
            const bbox = getBoundingBox(lat, lng, radiusKm);

            let airLabsResult = null;
            let openSkyResult = null;

            // 1. Chiamata ad AirLabs
            const airLabsUrl = `https://airlabs.co/api/v9/flights?api_key=${AIRLABS_API_KEY}&bbox=${bbox.lamin.toFixed(4)},${bbox.lomin.toFixed(4)},${bbox.lamax.toFixed(4)},${bbox.lomax.toFixed(4)}`;
            
            // 2. Chiamata ad OpenSky Network
            const openSkyUrl = `https://opensky-network.org/api/states/all?lamin=${bbox.lamin.toFixed(4)}&lomin=${bbox.lomin.toFixed(4)}&lamax=${bbox.lamax.toFixed(4)}&lomax=${bbox.lomax.toFixed(4)}`;

            // Eseguiamo le chiamate in parallelo gestendo eventuali fallimenti singoli
            const promises = [
                fetchWithBackoff(airLabsUrl).then(data => airLabsResult = data).catch(err => console.warn("AirLabs Radar fallito:", err)),
                fetchWithBackoff(openSkyUrl).then(data => openSkyResult = data).catch(err => console.warn("OpenSky Radar fallito:", err))
            ];

            await Promise.allSettled(promises);

            // Fusione e De-duplicazione dei dati
            allScannedFlights = mergeFlights(airLabsResult, openSkyResult, lat, lng, radiusKm);
            
            // Rendering grafico
            renderRadarFlights();

            loader.classList.add('hidden');
            iconReady.classList.remove('hidden');
        }

        // --- FUSIONE DEI DATI DELLE DUE FONTI ---
        function mergeFlights(airLabsData, openSkyData, centerLat, centerLng, maxRadius) {
            const flightMap = new Map();

            // Elaborazione AirLabs (Dati primari ricchi di informazioni sulla rotta)
            if (airLabsData && airLabsData.response) {
                airLabsData.response.forEach(item => {
                    if (!item.hex || !item.lat || !item.lng) return;
                    
                    const pLat = parseFloat(item.lat);
                    const pLng = parseFloat(item.lng);
                    const distance = getHaversineDistance(centerLat, centerLng, pLat, pLng);

                    // Filtra rigorosamente entro il raggio per precisione geometrica
                    if (distance > maxRadius) return;

                    const hexKey = item.hex.toLowerCase().trim();
                    flightMap.set(hexKey, {
                        hex: hexKey,
                        callsign: item.flight_iata || item.flight_icao || item.reg_number || 'N/D',
                        flight_iata: item.flight_iata || '',
                        flight_icao: item.flight_icao || '',
                        reg_number: item.reg_number || 'N/D',
                        flag: item.flag || 'N/D',
                        lat: pLat,
                        lng: pLng,
                        alt_m: item.alt ? Math.round(parseFloat(item.alt)) : null,
                        alt_ft: item.alt ? Math.round(parseFloat(item.alt) * 3.28084) : null,
                        dir: item.dir ? parseFloat(item.dir) : 0,
                        speed_kmh: item.speed ? Math.round(parseFloat(item.speed)) : null,
                        speed_kt: item.speed ? Math.round(parseFloat(item.speed) / 1.852) : null,
                        v_speed: item.v_speed ? parseFloat(item.v_speed) : 0,
                        squawk: item.squawk || 'N/D',
                        airline_icao: item.airline_icao || 'N/D',
                        airline_iata: item.airline_iata || 'N/D',
                        dep_iata: item.dep_iata || '',
                        arr_iata: item.arr_iata || '',
                        aircraft_icao: item.aircraft_icao || 'N/D',
                        status: item.status || 'active',
                        distance: distance,
                        source: 'AirLabs'
                    });
                });
            }

            // Elaborazione OpenSky (Completa eventuali dati mancanti o traccia voli non rilevati da AirLabs)
            if (openSkyData && openSkyData.states) {
                openSkyData.states.forEach(state => {
                    if (!state[0] || state[6] === null || state[5] === null) return;
                    
                    const pLat = parseFloat(state[6]);
                    const pLng = parseFloat(state[5]);
                    const distance = getHaversineDistance(centerLat, centerLng, pLat, pLng);

                    if (distance > maxRadius) return;

                    const hexKey = state[0].toLowerCase().trim();
                    const existing = flightMap.get(hexKey);

                    if (existing) {
                        // Unione intelligente: se esiste già, aggiorniamo telemetria e indichiamo doppia fonte
                        existing.source = 'AirLabs + OpenSky';
                        if (!existing.dir && state[10] !== null) existing.dir = parseFloat(state[10]);
                        if (!existing.alt_ft && state[7] !== null) {
                            existing.alt_m = Math.round(state[7]);
                            existing.alt_ft = Math.round(state[7] * 3.28084);
                        }
                        if (!existing.speed_kt && state[9] !== null) {
                            existing.speed_kmh = Math.round(state[9] * 3.6);
                            existing.speed_kt = Math.round(state[9] * 1.94384);
                        }
                        if (existing.squawk === 'N/D' && state[14]) existing.squawk = state[14];
                    } else {
                        // Nuovo volo trovato solo su OpenSky
                        const callsign = (state[1] || '').trim() || 'N/D';
                        const speed_ms = state[9] !== null ? parseFloat(state[9]) : 0;
                        const alt_m = state[7] !== null ? parseFloat(state[7]) : 0;

                        flightMap.set(hexKey, {
                            hex: hexKey,
                            callsign: callsign,
                            flight_iata: callsign,
                            flight_icao: callsign,
                            reg_number: 'N/D',
                            flag: countryToFlagEmoji(state[2]),
                            lat: pLat,
                            lng: pLng,
                            alt_m: Math.round(alt_m),
                            alt_ft: Math.round(alt_m * 3.28084),
                            dir: state[10] !== null ? parseFloat(state[10]) : 0,
                            speed_kmh: Math.round(speed_ms * 3.6),
                            speed_kt: Math.round(speed_ms * 1.94384),
                            v_speed: state[11] !== null ? parseFloat(state[11]) : 0,
                            squawk: state[14] || 'N/D',
                            airline_icao: 'N/D',
                            airline_iata: 'N/D',
                            dep_iata: '',
                            arr_iata: '',
                            aircraft_icao: 'N/D',
                            status: 'active',
                            distance: distance,
                            source: 'OpenSky'
                        });
                    }
                });
            }

            // Ordina la lista per distanza crescente dal centro radar
            return Array.from(flightMap.values()).sort((a, b) => a.distance - b.distance);
        }

        // Cliccando sul pulsante "Traccia"
        async function executeSingleTracking() {
            const query = document.getElementById('input-track-code').value.trim().toLowerCase();
            if (!query) {
                alertMessage("Inserisci un codice volo o codice ICAO24 HEX valido.");
                return;
            }

            showTrackerLoader(true);

            // Resetta mappa single
            singleFlightLayerGroup.clearLayers();

            try {
                let liveFlightData = null;

                // Proviamo a determinare se l'utente ha inserito un codice HEX (es. lungo 6 caratteri alfanumerici)
                // Altrimenti lo cerchiamo come IATA/ICAO flight code.
                let searchParam = "flight_iata";
                if (query.length === 6 && /^[a-f0-9]+$/i.test(query)) {
                    searchParam = "hex";
                } else if (query.startsWith("cs")) {
                    searchParam = "flight_icao";
                }

                const url = `https://airlabs.co/api/v9/flights?api_key=${AIRLABS_API_KEY}&${searchParam}=${encodeURIComponent(query.toUpperCase())}`;
                const data = await fetchWithBackoff(url);

                if (data && data.response && data.response.length > 0) {
                    liveFlightData = data.response[0];
                }

                // Se non troviamo una traccia live su AirLabs, proviamo con OpenSky per vedere se è rilevato in tempo reale come hex
                if (!liveFlightData && searchParam === "hex") {
                    const openSkySingleUrl = `https://opensky-network.org/api/states/all?icao24=${query}`;
                    const osData = await fetchWithBackoff(openSkySingleUrl);
                    if (osData && osData.states && osData.states.length > 0) {
                        const state = osData.states[0];
                        liveFlightData = {
                            hex: state[0],
                            reg_number: 'N/D',
                            flag: countryToFlagEmoji(state[2]),
                            lat: parseFloat(state[6]),
                            lng: parseFloat(state[5]),
                            alt: state[7] ? Math.round(state[7]) : null,
                            dir: state[10] ? parseFloat(state[10]) : 0,
                            speed: state[9] ? Math.round(state[9] * 3.6) : null,
                            squawk: state[14] || 'N/D',
                            status: 'active',
                            flight_iata: state[1] ? state[1].trim() : 'N/D',
                            airline_icao: 'N/D',
                            dep_iata: '',
                            arr_iata: ''
                        };
                    }
                }

                if (!liveFlightData) {
                    alertMessage("Volo attualmente non attivo o fuori copertura radar. Non è stato possibile localizzare le coordinate.");
                    showTrackerLoader(false);
                    return;
                }

                // Mostra e popola i dati
                await populateTrackerDashboard(liveFlightData);

            } catch (err) {
                alertMessage("Errore di rete o rate-limiting durante la query alle API. Riprova tra poco.");
                showTrackerLoader(false);
            }
        }

        // --- RISOLUTORE DETTAGLI AEROPORTO CON CACHING ---
        async function fetchAirportData(iataCode) {
            const cleanIata = iataCode.toUpperCase().trim();
            if (!cleanIata || cleanIata.length !== 3) return null;

            if (resolvedAirportsCache[cleanIata]) {
                return resolvedAirportsCache[cleanIata];
            }

            try {
                const url = `https://airlabs.co/api/v9/airports?api_key=${AIRLABS_API_KEY}&iata_code=${cleanIata}`;
                const data = await fetchWithBackoff(url);

                if (data && data.response && data.response.length > 0) {
                    const ap = data.response[0];
                    const airportObj = {
                        name: ap.name || 'Aeroporto Sconosciuto',
                        city: ap.city || ap.name || 'Città non disp.',
                        country_code: ap.country_code || '',
                        lat: ap.lat,
                        lng: ap.lng
                    };
                    resolvedAirportsCache[cleanIata] = airportObj;
                    return airportObj;
                }
            } catch (err) {
                console.warn(`Errore durante la risoluzione dell'aeroporto ${cleanIata}:`, err);
            }
            return null;
        }

        // --- MATEMATICA: FORMULA GEODETICA HAVERSINE (DISTANZA IN KM) ---
        function getHaversineDistance(lat1, lon1, lat2, lon2) {
            const R = 6371; // Raggio medio terrestre in Km
            const dLat = (lat2 - lat1) * Math.PI / 180;
            const dLon = (lon2 - lon1) * Math.PI / 180;
            const a =
                Math.sin(dLat / 2) * Math.sin(dLat / 2) +
                Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
                Math.sin(dLon / 2) * Math.sin(dLon / 2);
            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
            return R * c; 
        }

        function countryToFlagEmoji(country) {
            if (!country || country === 'N/D') return "🏳️";
            if (country.length === 2) {
                // Se è già un codice ISO a 2 lettere
                const codePoints = country.toUpperCase().split('').map(char =>  127397 + char.charCodeAt(0));
                return String.fromCodePoint(...codePoints);
            }
            const nameLower = country.toLowerCase().trim();
            for (const [key, emoji] of Object.entries(countryToFlagMap)) {
                if (nameLower.includes(key)) return emoji;
            }
            return "🏳️";
        }
