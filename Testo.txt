        // Formula matematica di Haversine per calcolare la distanza tra due punti geografici
        function getDistance(lat1, lon1, lat2, lon2) {
            const R = 6371; // Raggio della Terra in km
            const dLat = (lat2 - lat1) * Math.PI / 180;
            const dLon = (lon2 - lon1) * Math.PI / 180;
            const a = 
                Math.sin(dLat/2) * Math.sin(dLat/2) +
                Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2);
            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
            return R * c;
        }

        // Calcola il Bounding Box geografico attorno al centro in base al raggio scelto
        function getBoundingBox(centerLat, centerLng, radiusKm) {
            // Conversione approssimativa ma accurata per l'uso radar:
            // 1 grado di latitudine ≈ 111 km
            // 1 grado di longitudine ≈ 111 km * cos(latitudine)
            const latDelta = radiusKm / 111.0;
            const lngDelta = radiusKm / (111.0 * Math.cos(centerLat * Math.PI / 180));

            return {
                lamin: centerLat - latDelta,
                lamax: centerLat + latDelta,
                lomin: centerLng - lngDelta,
                lomax: centerLng + lngDelta
            };
        }

        // Chiamata AJAX con fetch e gestione errori/retry con esponenziale backoff
        async function fetchPlaneData() {
            const bbox = getBoundingBox(state.center.lat, state.center.lng, state.radius);
            const url = `https://opensky-network.org/api/states/all?lamin=${bbox.lamin}&lomin=${bbox.lomin}&maxlamin=${bbox.lamax}&maxlomin=${bbox.lomax}`;
            
            document.getElementById('status-text').textContent = "Aggiornamento radar in corso...";

            let success = false;
            let attempts = 0;
            let delay = 1000; // Inizio backoff a 1s

            while (!success && attempts < 5) {
                try {
                    const response = await fetch(url);
                    if (!response.ok) {
                        throw new Error(`Errore Server: ${response.status}`);
                    }
                    const data = await response.json();
                    processPlaneStates(data.states || []);
                    success = true;
                    
                    const time = new Date().toLocaleTimeString();
                    document.getElementById('status-text').textContent = `Ultimo segnale ADS-B: ${time}`;
                } catch (error) {
                    attempts++;
                    if (attempts >= 5) {
                        showToast("Impossibile connettersi ai server OpenSky. Riproverò tra poco.", "error");
                        document.getElementById('status-text').textContent = "Connessione API fallita.";
                    } else {
                        await new Promise(resolve => setTimeout(resolve, delay));
                        delay *= 2; // Raddoppia ad ogni tentativo fallito
                    }
                }
            }
        }

        // Elabora l'array di stati aerei ricevuti dall'API
        function processPlaneStates(statesArray) {
            const activeIcaos = new Set();

            statesArray.forEach(arr => {
                // Mappatura parametri OpenSky
                // Informazioni dettagliate: https://openskynetwork.github.io/opensky-api/rest.html#response
                const plane = {
                    icao24: arr[0],
                    callsign: arr[1] ? arr[1].trim() : "UNKN",
                    country: arr[2] || "Sconosciuto",
                    lng: arr[5],
                    lat: arr[6],
                    baroAltitude: arr[7] !== null ? Math.round(arr[7]) : null, // metri
                    velocity: arr[9] !== null ? Math.round(arr[9] * 3.6) : 0,  // m/s -> km/h
                    heading: arr[10] !== null ? Math.round(arr[10]) : 0,        // gradi rotta
                    verticalRate: arr[11] !== null ? arr[11] : 0,             // m/s salita/discesa
                    geoAltitude: arr[13] !== null ? Math.round(arr[13]) : null, // metri
                    onGround: arr[8]
                };

                // Filtro supplementare: Verifichiamo che l'aereo rientri effettivamente nel raggio circolare reale
                if (plane.lat && plane.lng) {
                    const distance = getDistance(state.center.lat, state.center.lng, plane.lat, plane.lng);
                    if (distance <= state.radius) {
                        activeIcaos.add(plane.icao24);
                        updateOrCreatePlaneMarker(plane);
                    }
                }
            });

            // Rimuovi aerei scomparsi (non più presenti nel bounding box dell'API)
            for (const icao24 in state.planeMarkers) {
                if (!activeIcaos.has(icao24)) {
                    // Non rimuovere se è l'aereo che stiamo tracciando in modo intensivo (fintanto che siamo in tracking tab)
                    if (icao24 === state.trackingTarget && state.activeTab === 'tracking') {
                        continue;
                    }
                    
                    state.map.removeLayer(state.planeMarkers[icao24].marker);
                    delete state.planeMarkers[icao24];
                }
            }

            // Se stiamo tracciando attivamente un aereo, aggiorna il pannello informativo con i dati più freschi
            if (state.trackingTarget && state.planeMarkers[state.trackingTarget]) {
                updateInfoPanel(state.planeMarkers[state.trackingTarget].data);
            }
        }

        // Disegna o aggiorna dinamicamente i marker aerei (Fase C)
        function updateOrCreatePlaneMarker(plane) {
            const latlng = [plane.lat, plane.lng];
            const isTracking = (plane.icao24 === state.trackingTarget);
            
            // Colore differente se l'aereo è quello sotto tracciamento attivo
            const markerColor = isTracking ? '#e11d48' : '#10b981';

            const planeIcon = L.divIcon({
                html: svgIcons.plane(plane.heading, markerColor),
                className: 'plane-marker-icon-container',
                iconSize: [30, 30],
                iconAnchor: [15, 15]
            });

            if (state.planeMarkers[plane.icao24]) {
                // Spostamento fluido della posizione
                state.planeMarkers[plane.icao24].marker.setLatLng(latlng);
                state.planeMarkers[plane.icao24].marker.setIcon(planeIcon);
                state.planeMarkers[plane.icao24].data = plane;
            } else {
                const marker = L.marker(latlng, { icon: planeIcon }).addTo(state.map);
                
                // Evento click sul marker aereo
                marker.on('click', () => {
                    selectPlane(plane.icao24);
                });

                state.planeMarkers[plane.icao24] = { marker, data: plane };
            }

            // Aggiorna lo storico e la linea di scia (polilinea) se l'aereo è il target di tracciamento
            if (isTracking && state.activeTab === 'tracking') {
                addToTrackingPath(latlng);
            }
        }

        // Seleziona l'aereo e mostra il pannello info rapido (Fase D)
        function selectPlane(icao24) {
            const planeData = state.planeMarkers[icao24].data;
            updateInfoPanel(planeData);
            
            // Mostra il pannello informativo
            const infoPanel = document.getElementById('info-panel');
            infoPanel.classList.remove('hidden');

            // Salva l'aereo selezionato temporalmente
            state.tempSelectedIcao = icao24;
        }

        function updateInfoPanel(data) {
            document.getElementById('plane-callsign').textContent = data.callsign || "Sconosciuto";
            document.getElementById('plane-icao').textContent = data.icao24.toUpperCase();
            document.getElementById('plane-country').textContent = data.country;
            document.getElementById('plane-speed').textContent = `${data.velocity} km/h`;
            document.getElementById('plane-altitude').textContent = data.geoAltitude !== null ? `${data.geoAltitude} m` : "Non disp.";
            document.getElementById('plane-baro').textContent = data.baroAltitude !== null ? `${data.baroAltitude} m` : "Non disp.";
            document.getElementById('plane-heading').textContent = `${data.heading}°`;
            document.getElementById('plane-ground').textContent = data.onGround ? "Al Suolo" : "In Volo";
            
            // Velocità verticale con indicazione salita/discesa
            const vRate = data.verticalRate;
            let vRateText = `${vRate} m/s`;
            if (vRate > 0.5) vRateText += " (Salita ↗)";
            else if (vRate < -0.5) vRateText += " (Discesa ↘)";
            document.getElementById('plane-vertical-rate').textContent = vRateText;
        }

        // Chiude il pannello informativo
        function closeInfoPanel() {
            document.getElementById('info-panel').classList.add('hidden');
        }


        // ==========================================
        // TRANSIZIONE E SCHEDA TRACCIAMENTO (FASE D)
        // ==========================================

        function enableAdvancedTracking() {
            const icao24 = state.tempSelectedIcao;
            if (!icao24 || !state.planeMarkers[icao24]) return;

            // Pulisci il vecchio tracciamento se attivo
            clearTrackingData();

            state.trackingTarget = icao24;
            const targetData = state.planeMarkers[icao24].data;

            // Cambia stato bottone Navbar Tracciamento
            const tabTrackingBtn = document.getElementById('tab-tracking');
            tabTrackingBtn.removeAttribute('disabled');

            // Passa alla scheda tracciamento
            activateTab('tracking');

            // Centra la mappa sull'aereo
            state.map.setView([targetData.lat, targetData.lng], 10);

            // Inizializza storico coordinate per polilinea
            state.trackingHistory = [[targetData.lat, targetData.lng]];
            state.trackingPath = L.polyline(state.trackingHistory, {
                color: '#ef4444',
                weight: 3,
                opacity: 0.8,
                dashArray: '5, 5'
            }).addTo(state.map);

            showToast(`Tracciamento avanzato attivato per volo ${targetData.callsign}`, "success");
        }

        function addToTrackingPath(latlng) {
            // Verifica che l'ultima coordinata sia diversa per evitare duplicati statici
            const lastCoord = state.trackingHistory[state.trackingHistory.length - 1];
            if (!lastCoord || lastCoord[0] !== latlng[0] || lastCoord[1] !== latlng[1]) {
                state.trackingHistory.push(latlng);
                if (state.trackingPath) {
                    state.trackingPath.setLatLngs(state.trackingHistory);
                }
                
                // Segui l'aereo tracciato muovendo la mappa
                state.map.panTo(latlng);
            }
        }

        function clearTrackingData() {
            if (state.trackingPath) {
                state.map.removeLayer(state.trackingPath);
                state.trackingPath = null;
            }
            state.trackingHistory = [];
            state.trackingTarget = null;
        }

        // Navigazione interna alle tab dell'App
        function activateTab(tabId) {
            state.activeTab = tabId;

            // Elementi grafici dei bottoni tab
            document.getElementById('tab-radar').classList.remove('active');
            document.getElementById('tab-tracking').classList.remove('active');
            document.getElementById(`tab-${tabId}`).classList.add('active');

            if (tabId === 'radar') {
                // Torna al Radar Globale
                // Rendi visibili tutti gli aerei e gli elementi dell'overlay radar
                if (state.radarCircle) {
                    state.radarCircle.addTo(state.map);
                    state.map.fitBounds(state.radarCircle.getBounds());
                }
                
                // Ripristina colorazioni normali
                for (const icao in state.planeMarkers) {
                    const plane = state.planeMarkers[icao].data;
                    const normalIcon = L.divIcon({
                        html: svgIcons.plane(plane.heading, '#10b981'),
                        className: 'plane-marker-icon-container',
                        iconSize: [30, 30],
                        iconAnchor: [15, 15]
                    });
                    state.planeMarkers[icao].marker.setIcon(normalIcon);
                }
                
                clearTrackingData();
                document.getElementById('tab-tracking').setAttribute('disabled', 'true');
                closeInfoPanel();
                
            } else if (tabId === 'tracking') {
                // Siamo nella tab tracciamento
                // Rimuovi temporaneamente la visualizzazione del cerchio radar per isolare la visuale
                if (state.radarCircle) {
                    state.map.removeLayer(state.radarCircle);
                }
                
                // Isola ed evidenzia l'aereo target
                if (state.trackingTarget && state.planeMarkers[state.trackingTarget]) {
                    const tData = state.planeMarkers[state.trackingTarget].data;
                    
                    const highlightedIcon = L.divIcon({
                        html: svgIcons.plane(tData.heading, '#ef4444'),
                        className: 'plane-marker-icon-container',
                        iconSize: [32, 32],
                        iconAnchor: [16, 16]
                    });
                    state.planeMarkers[state.trackingTarget].marker.setIcon(highlightedIcon);
                    state.map.setView([tData.lat, tData.lng], 10);
                }
            }
        }


        // ==========================================
        // COMPONENTI DI NOTIFICA E UTILITY UI
        // ==========================================

        function showToast(message, type = "error") {
            const notif = document.getElementById('custom-notification');
            const text = document.getElementById('notification-message');
            
            text.textContent = message;
            
            // Adatta colore in base al tipo
            if (type === "success") {
                notif.style.background = "rgba(16, 185, 129, 0.95)";
            } else {
                notif.style.background = "rgba(239, 68, 68, 0.95)";
            }

            notif.classList.add('show');

            setTimeout(() => {
                notif.classList.remove('show');
            }, 4000);
        }
    </script>
